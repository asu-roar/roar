cmake_minimum_required(VERSION 3.0.2)
project(supervisor)


find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  roar_msgs
)


catkin_package(
  CATKIN_DEPENDS
  roscpp
  rospy
  std_msgs
  roar_msgs
)


include_directories(
  ${catkin_INCLUDE_DIRS}
)


 catkin_install_python(PROGRAMS
   src/auto_supervisor.py
   src/roar_supervisor.py
   test/manual.py
   test/autonomous.py
   test/mode_sender.py
   DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
 )
